\section{Home Assignment 9: Connections and curvature}
\setcounter{section}{9}

\begin{exercise}
	Let $(B,\nabla )$ be a rank 1 bundle with connection over a circle $S^{1}$, and $\nabla ':=\nabla +\theta$, where $\theta\in \Lambda^{1}(S^{1})$ is a 1-form, understood as a section of $\Lambda^{1}(S^{1})\otimes \operatorname{End}(B)$. Prove that holonomy of $\nabla '$ is equal to holonomy of $\nabla$ if and only if $\theta$ is exact.
\end{exercise}

\begin{exercise}
Let $(B,\nabla )$ be a bundle with connection over $M$, and $\nabla ':=\nabla +\theta$, where $\theta\in \Lambda^{1}(M)$ is a 1-form, understood as a section of $\Lambda^{1}(S^{1})\otimes \operatorname{Id}_{B}$, where $\operatorname{Id}_{B}\in \operatorname{End}(B)$ is the identity section. Assume that both $\nabla$ and $\nabla '$ are flat. Prova that $d\theta=0$.
\end{exercise}

\begin{exercise}
	Let $B$ be a trivial real vector bundle over a circle $S^{1}$, $x\in S^{1}$ a point, and $A\in \operatorname{Aut}(B|_{x})$ a linear automorphism with all eigenvalues real and positive, or all eigenvalues complex.Construct a connection $\nabla$ such that its holonomy in $x$ is equal to $A$.
\end{exercise}

\begin{exercise}
	Let $B$ be a rank 1 bundle on $\mathbb{R}^2$, and $V \in \Lambda^{2}(\mathbb{R}^{2})$ a 2-form. Construct a connection on $B$ such that its curvature is equal to $V$.
\end{exercise}

\begin{exercise}
	Let $B$ be a bundle on $\mathbb{R}^{2}$, and $R\in \Lambda^{2}(T^{*}_{0}\mathbb{R}^{n})\times \operatorname{End}(B|_{0})$ a 2-form on the tangent vector space in $0$ with values in $\operatorname{End}(B|_{0})$. Construct a connection with curvature $\Theta$ such that $\Theta|_{0}=R$.
	\begin{enumerate}[label=\alph*.]
		\item When $n=2$, and $R$ takes values in traceless endomorphisms.
		\item When $n=2$, and $R$ arbitrary.
		\item For arbitrary $n$ and $R$.
	\end{enumerate}
\end{exercise}
